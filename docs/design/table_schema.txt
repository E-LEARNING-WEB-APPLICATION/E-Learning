auth
-----
auth_id       INT PRIMARY KEY AUTO_INCREMENT
email         VARCHAR(255) UNIQUE NOT NULL
password_hash VARCHAR(255) NOT NULL
role          ENUM('student','instructor','admin') NOT NULL
status        ENUM('active','inactive') DEFAULT 'active'
created_at    DATETIME DEFAULT CURRENT_TIMESTAMP


user_profile
-------------
profile_id   INT PRIMARY KEY AUTO_INCREMENT
auth_id      INT UNIQUE NOT NULL REFERENCES auth(auth_id)   -- 1:1
fname        VARCHAR(50)
lname        VARCHAR(50)
dob          DATE
gender       ENUM('male','female','other')
profile_pic  VARCHAR(255)
created_at   DATETIME DEFAULT CURRENT_TIMESTAMP


student_details
----------------
sID           INT PRIMARY KEY AUTO_INCREMENT
profile_id    INT UNIQUE NOT NULL REFERENCES user_profile(profile_id)  -- 1:1
add_id        INT REFERENCES address(add_id)
created_at    DATETIME DEFAULT CURRENT_TIMESTAMP


instructor_details
-------------------
iId           INT PRIMARY KEY AUTO_INCREMENT
profile_id    INT UNIQUE NOT NULL REFERENCES user_profile(profile_id)  -- 1:1
specialization   VARCHAR(100)
bio           TEXT
experience    VARCHAR(50)
mob_no        VARCHAR(15)
status        ENUM('active','inactive') DEFAULT 'active'
isApproved    ENUM('pending' , 'approved' , 'rejected') DEFAULT 'pending'
add_id        INT REFERENCES address(add_id)
created_at    DATETIME DEFAULT CURRENT_TIMESTAMP


education_details
------------------
eid            INT PRIMARY KEY AUTO_INCREMENT
student_id     INT REFERENCES student_details(sID)           -- 1 student â†’ many education entries
instructor_id  INT REFERENCES instructor_details(iId)        -- optional for instructors
degree         VARCHAR(100)
field_of_study VARCHAR(100)
institute      VARCHAR(100)
grade          VARCHAR(10)
description    TEXT
start_date     DATE
end_date       DATE
created_at     DATETIME DEFAULT CURRENT_TIMESTAMP


address
--------
add_id        INT PRIMARY KEY AUTO_INCREMENT
address_line1 VARCHAR(255)
address_line2 VARCHAR(255)
city          VARCHAR(50)
state         VARCHAR(50)
pincode       VARCHAR(10)
country       VARCHAR(50)
created_at    DATETIME DEFAULT CURRENT_TIMESTAMP


category
--------
cId           INT PRIMARY KEY AUTO_INCREMENT
cname         VARCHAR(50)
cdesc         TEXT
created_at    DATETIME DEFAULT CURRENT_TIMESTAMP


courses
--------
courseId           INT PRIMARY KEY AUTO_INCREMENT
c_id               INT REFERENCES category(cId)
iId                INT REFERENCES instructor_details(iId)
course_name        VARCHAR(100)
course_desc        TEXT
fees               DECIMAL(10,2)
discount_percentage DECIMAL(5,2)
course_thumbnail   VARCHAR(255)
course_intro_video VARCHAR(255)
course_duration    VARCHAR(50)
created_at         DATETIME DEFAULT CURRENT_TIMESTAMP



sections
--------
sId             INT PRIMARY KEY AUTO_INCREMENT
course_id       INT REFERENCES courses(courseId)
section_name    VARCHAR(100)
section_desc    TEXT
created_at      DATETIME DEFAULT CURRENT_TIMESTAMP


topics
------
t_id           INT PRIMARY KEY AUTO_INCREMENT
s_id           INT REFERENCES sections(sId)
topic_name     VARCHAR(100)
topic_desc     TEXT
topic_video    VARCHAR(255)
topic_notes    TEXT
created_at     DATETIME DEFAULT CURRENT_TIMESTAMP


booking
-------
booking_id       INT PRIMARY KEY AUTO_INCREMENT
student_id       INT REFERENCES student_details(sID)
course_id        INT REFERENCES courses(courseId)
payment_id       INT REFERENCES transactions(tid)
date_of_purchase DATETIME DEFAULT CURRENT_TIMESTAMP
course_price     DECIMAL(10,2)   -- snapshot of course price at booking
price_paid       DECIMAL(10,2)   -- actual amount student paid


transactions
-------------
tid             INT PRIMARY KEY AUTO_INCREMENT
mode_of_payment VARCHAR(50)
amount          DECIMAL(10,2)
created_at      DATETIME DEFAULT CURRENT_TIMESTAMP


feedback
--------
feedback_id     INT PRIMARY KEY AUTO_INCREMENT
student_id      INT REFERENCES student_details(sID)
course_id       INT REFERENCES courses(courseId)
rating          DECIMAL(2,1)   -- e.g., 4.5 out of 5
comment         TEXT
created_at      DATETIME DEFAULT CURRENT_TIMESTAMP
